  es.forEach(e=>{ if(e.isIntersecting){e.target.classList.add('in');obs.unobserve(e.target);} });
},{threshold:.08});
document.querySelectorAll('.r').forEach(el=>obs.observe(el));

/* --- Counter animation --- */
function countUp(el){
  const target=+el.dataset.count;
  const suffix=el.dataset.suffix||'';
  let val=0, dur=1600, fps=60;
  const step=target/(dur/1000*fps);
  const t=setInterval(()=>{
    val+=step;
    if(val>=target){el.textContent=target+suffix;clearInterval(t);}
    else el.textContent=Math.floor(val)+suffix;
  },1000/fps);
}
const cObs=new IntersectionObserver(es=>{
  es.forEach(e=>{
    if(e.isIntersecting){
      e.target.querySelectorAll('[data-count]').forEach(countUp);
      cObs.unobserve(e.target);
    }
  });
},{threshold:.4});
const sg=document.querySelector('.stats-g');
if(sg) cObs.observe(sg);

    tab.classList.add('active');
    document.querySelectorAll('.proj-panel').forEach(p=>p.classList.remove('active'));
    const panel=document.getElementById('panel-'+key);
    if(panel){
      panel.classList.add('active');
      panel.querySelectorAll('.r:not(.in)').forEach(el=>obs.observe(el));
    }
  });
});
